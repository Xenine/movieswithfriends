<h2>Стек технологий: </h2>
<ul>
  <li>Backend
    <ul> 
      <li>Django</li>
      <li>DRF</li>
      <li>PostgreSQL</li>
      <li>Celery</li>
    </ul>
  </li>
  <li>Frontend
    <ul> 
      <li>React</li>
      <li>MobX</li>
      <li>Sass, CSS modules</li>
      <li>PWA</li>
    </ul>
  </li>
</ul>
